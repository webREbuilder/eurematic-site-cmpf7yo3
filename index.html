Technical Flow

How Klara Works

⌚
1

Wearable Connect

User authorizes Klara to securely read from Oura, Whoop, or Apple HealthKit via encrypted off-chain tunnels.

🤖
2

AI Processing

Our ML pipeline analyzes biometrics off-chain, distilling raw data into structured wellness achievement signals.

🔐
3

DP Proof Generation

Differential privacy is applied. A cryptographic ZK-style attestation is generated — verifiable but not reversible.

🏆

On-Chain Attestation

The proof is published to EVM chains. Your protocol smart contract reads it to unlock rewards, discounts, or access rights.

WellnessProtocol.sol 
import "@klara-oracle/contracts/KlaraConsumer.sol";

contract FitFiRewards is KlaraConsumer {

  function claimReward(bytes32 userId) external {
    // Get privacy-preserved attestation
    KlaraAttestation memory proof = 
      getAttestation(userId);

    // Verify without seeing raw biometrics
    require(proof.achievementVerified, "Not verified");
    require(proof.timestamp > block.timestamp - 1 days);

    // Safe to reward — no PII ever on-chain
    _mintRewardTokens(msg.sender, DAILY_REWARD);
  }
}
